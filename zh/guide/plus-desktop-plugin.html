        # 开始使用

        在页面引入 mui-player-mobile-plugin.js，插件需要在初始化播放器之前加载：

        <!-- 使用 script 标签引入 -->
<script type="text/javascript" src="js/mui-player-mobile-plugin.min.js"></script>

        在主配置项 plugins 中传入：

        var mp = new MuiPlayer({
    container:'#mui-player',
    src:'../media/media.mp4',
    ...
    
    plugins:[
        new MuiPlayerMobilePlugin({
            key: '插件密文',
            showMenuButton:true,
            ...
        })
    ]
});

        # Plus 配置选项

        示例创建一个弹幕输入组件：

        new MuiPlayerMobilePlugin({
    key: '插件密文',
    danmakuComponent:{
        open: true, // 是否开启弹幕组件
        growSize: 0.65, // 组件占据容器的宽度比例，取值范围：0 - 1
        buttonText: '发送',
        onSend(text, cleanText) { // 监听文本发送内容的函数回调
            // axtion
        },
    }
})
        属性名 类型 默认值 说明
        danmakuComponent Object {} 创建一个弹幕输入组件
        danmakuComponent.open Object true 是否开启弹幕组件
        danmakuComponent.growSize Number 0.65 组件占据容器的宽度比例，取值范围：0 - 1
        danmakuComponent.position String left 组件排列的位置，可取值 left | right | center
        danmakuComponent.buttonText String - 按钮显示的文字
        danmakuComponent.placeholder String - 弹幕文本录入框显示的占位符
        danmakuComponent.onSend Function (text,cleanText) 监听文本发送内容的函数回调，方法接收两个回调参数：
        text：String - 表示文本内容
        cleanText：Function - 调用函数清空文本输入框内容
        danmakuComponent.loadIntercept Function (danmakuControls) 在组件加载之前进行拦截，通常这里是做对组件重写的动作：danmakuControls = Element
        danmakuComponent.style Object {} 组件的附加样式
        #
        fullScalingControls Array [] 指定全屏进行缩放控件的元素

        # Plus 接口

        方法名 类型 默认值 必填 说明
        PopupAdvertise Function (options) 弹出广告悬浮窗
        PopupAdvertise.type String image-advertising -
        PopupAdvertise.image String Y 图片地址
        style Object {} 自定义附加弹出窗样式
        click Function 广告悬浮窗点击后触发
        onerror Function 图片地址加载错误时触发
        #
        addWatermark Function (options) 添加窗口播放水印，参考
        addWatermark.image String Y logo 水印地址
        addWatermark.scale Number 1 水印显示的缩放比
        position String top-right 水印显示的位置，可选：top-right | top-left | bottom-left | bottom-right
        #
        createRolling Function (options) 创建一个播放跑马灯窗口
        createRolling.id String "" "" 滚动播放的ID标识，如需创建多个滚动窗口该参数必须声明
        createRolling.scrollamount Number 6 设置每次滚动时移动的长度（以像素为单位），默认值为 6
        createRolling.scrolldelay Number 85 设置每次滚动时的时间间隔（以毫秒为单位），默认值为 85
        createRolling.loop Number -1 设置滚动次数，默认值为 −1，表示将连续滚动
        createRolling.style Object {} 自定义附加跑马灯样式
        createRolling.onstart Function 开始滚动时触发
        createRolling.onfinish Function 滚动结束后触发，当 loop != -1 时有效
        #
        createSettingPopup Function (options) 创建并弹出一个配置设置窗口
        createSettingPopup.name String Y 弹窗标识ID
        createSettingPopup.configs Array Y 配置项目
        createSettingPopup.offseDistance Number 0 指定 popup 偏移距离
        #
        screenshot Function (options) 获取当前播放时间的视频截图，方法接收函数的可配置对象，参考

        # 创建视频贴片水印：

        mp.addWatermark({
    image: 'https://muiplayer.oss-cn-shanghai.aliyuncs.com/static/image/logot.png',
    scale: 0.5,
    position:'top-right',
})

        # 获取当前播放时间的视频截图：

        /**
* 获取当前播放时间的视频截图
* 注意: 通常一些 HTML元素提供了对 Cors 的支持，例如：<img>、<audio>、<video> 标签等，这意味着调用 screenshot 截图方法可能会产生跨域从行为不被支持！请设置元素的 crossOrigin 属性
* 参考：https://developer.mozilla.org/zh-CN/docs/Web/HTML/Attributes/crossorigin
*/
mp.screenshot({
    width: video.offsetWidth, // 视频截图画布的宽度【Type: Number, Default: video.offsetWidth】
    height: video.offsetHeight, // 视频画布截图的高度【Type: Number, Default: video.offsetHeight】
    quality: 0.92, // 设置 jpegs 图片输出的质量，取值 0 - 1【Type: Number, Default: 0.92】
    type: 'image/png', // 输出的图片格式，可选：image/jpeg | image/png | image/webp【Default: image/png】
	format: '.png', // 指定输出的图片格式，为空时默认根据 type 自动判断【Default: .png】
    scale:1, // 截图渲染的缩放比例，取值 0 - 1【Type: Number, Default: window.devicePixelRatio】
    onDone:function(data) { // 完成后触发的回调函数
        // base64: 截图生成的 base64 编码
        // currentTime: 当前播放的时间，以秒为单位；
        // download: 函数，参数传递一个 filename 下载截图文件
        let { base64, currentTime, download } = data;

        // document.getElementById('my-image').setAttribute('src', base64);
        // 下载截图文件
        download('screenshot file name');
    },
    onFail:function(error) { // 失败后触发的回调函数
        // console.error(error);
    }
});
